def berechne_kfa_praezise(groesse_m, gewicht_kg, bauchumfang_cm): ideal_gewicht = (groesse_m ** 2) * 23 diff = abs(gewicht_kg - ideal_gewicht) korrektur_wert = (diff / 3) * 0.75 if gewicht_kg >= ideal_gewicht: basis_kfa = 20 - korrektur_wert else: basis_kfa = 20 + korrektur_wert # Fehler behoben: Einrückung korrigiert, damit diese Variablen immer berechnet werden bauch_referenz = (groesse_m * 100) / 2 bauch_diff = bauchumfang_cm - bauch_referenz bauch_korrektur = (bauch_diff / 2) * 1.5 finaler_kfa = basis_kfa + bauch_korrektur return round(finaler_kfa, 2) # Daten für die Berechnung: ergebnis = berechne_kfa_praezise(1.70, 69, 79) print(f"Das Ergebnis für 1.70m, 69kg, 79cm Taille ist: {ergebnis}%")
